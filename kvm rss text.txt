1. sudo grep -c "svm\|vmx" /proc/cpuinfo
2. sudo apt-get update
3. sudo apt-get install qemu-kvm libvirt-bin bridge-utils virt-manager
4. sudo adduser naveen
6. sudo addgroup libvirtd
7. sudo adduser naveen libvirtd
8. sudo virsh -c qemu:///system list
9. sudo virt-manager  - open virtual machine
10. create new virtual machine
11. select os iso file location
12. follow other steps as normal installation.
